create table if not exists votes (
  id bigint generated by default as identity primary key,
  match_id text not null,
  player text not null,
  pick text not null,
  created_at timestamptz default now(),
  unique(match_id, player)
);

create table if not exists results (
  match_id text primary key,
  winner text not null,
  updated_at timestamptz default now()
);

alter table votes enable row level security;
alter table results enable row level security;

drop policy if exists "Votes are visible" on votes;
create policy "Votes are visible" on votes for select using (true);

drop policy if exists "Family can insert votes" on votes;
create policy "Family can insert votes" on votes for insert with check (true);

drop policy if exists "Results are visible" on results;
create policy "Results are visible" on results for select using (true);

drop policy if exists "Admin can save results" on results;
create policy "Admin can save results" on results for insert with check (true);

drop policy if exists "Admin can update results" on results;
create policy "Admin can update results" on results for update using (true) with check (true);
